本地训练测试

模型b

pip install pycocotools tensorboard
cd container/mask_r_cnn
python local_train.py \
  --root_train_data='../../data/modelc' \
  --model_type='modelc_sub' \
  --num_epochs=2 \
  --save_path='../../models/modelc'
